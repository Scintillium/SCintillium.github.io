    這道題雖然並不難，但我覺得學習研究，在於其有多種解法，且都是常用的思路，卻都能應用在同一個問題上。這種解法靈活的算法題非常適合我們來一次性複習各種解題思路以及他們的應用和寫法。

    問題描述

    鏈接

    • 718. 最长重复子数组
      兩個整數數組nums1和nums2，返回兩個數組中公共的，長度最長的子數組的長度。

    划動窗口解法

    一種直觀的解決方法就是，我們假想兩個數組是我們日常使用的兩把直尺，然後尺子上的刻度格子是數組，然後我們滑動兩把直尺將刻度線對齊。在重疊的部分查詢最長的重複子串。假設數組重複的部分起始下標分別爲start1和start2，則求重疊部分的最長重複子數組的函數可寫爲：

    int maxLength(vector<int>& nums1, int start1, vector<int>& nums2, int start2){
        int res = 0; // 記錄最長的重複子數組長度
        int cur = 0; // 以當前位置爲末位的最長重複子數組長度
        int i = 0;  // 重疊部分的相對偏移量來遍歷重疊部分
        while(start1 + i < nums1.size() && start2 + i < nums2.size()){    //循環條件
            if(nums1[start1+i] == nums2[start2+i])++cur;                  
            else cur = 0;
            res = max(res,cur);                                           //記錄最長的記錄
            ++i;
        }
        return res;
}

    假設數組nums1和nums2的長度分別爲N和M，在滑動兩個數組的時候可以發現，start1和start2的取值存在兩種狀況，一種是start1固定爲0，而start2的取值爲0M；第二種情況則是start2爲0，而start1的取值是0N。這樣我們就可以很容易的實現求最長重複子數組：

    class Solution {
public:
    int findLength(vector<int>& nums1, vector<int>& nums2) {
        int res = 0, n = nums1.size(), m = nums2.size();
        for(int i = 0; i < n; ++i){
            res = max(res, maxLength(nums1,i,nums2,0));
        }
        for(int i = 0; i < m; ++i){
            res = max(res, maxLength(nums1,0,nums2,i));
        }
        return res;
    }
    int maxLength(vector<int>& nums1, int start1, vector<int>& nums2, int start2){
        int res = 0;
        int cur = 0;
        int i = 0;
        while(start1 + i < nums1.size() && start2 + i < nums2.size()){
            if(nums1[start1+i] == nums2[start2+i])++cur;
            else cur = 0;
            res = max(res,cur);
            ++i;
        }
        return res;
    }
};

    在上述方法當中，我們可以知道兩個數組的重疊區域的長度先是從0遞增到min(N,M)，然後再遞減爲0；於是乎就很可能出現這樣一種情況，既先前所求到的最長重複子數組的長度會大於當前的數組重疊區域的長度，代表當前和後序的搜索都是不必要的了，即可以直接終止循環，返回結果。

    class Solution {
public:
    int findLength(vector<int>& nums1, vector<int>& nums2) {
        int res = 0, n = nums1.size(), m = nums2.size(), len= 0;
        for(int i = 0; i < n; ++i){
            res = max(res, maxLength(nums1,i,nums2,0));
        }
        for(int i = 0; i < m; ++i){
            len = min(n,m-i);
            if(len<= res)break;
            res = max(res, maxLength(nums1,0,nums2,i));
        }
        return res;
    }
    int maxLength(vector<int>& nums1, int start1, vector<int>& nums2, int start2){
        int res = 0;
        int cur = 0;
        int i = 0;
        while(start1 + i < nums1.size() && start2 + i < nums2.size()){
            if(nums1[start1+i] == nums2[start2+i])++cur;
            else cur = 0;
            res = max(res,cur);
            ++i;
        }
        return res;
    }
};

    時間複雜度： O(M+N) * min(M,N)

    空間複雜度: O(1)

    動態規劃

    在滑動窗口當中，兩個數組中的元素需要經過重複的比較。在這種需要減少重複步驟，可以利用一個二維數組DP來記錄中間狀態。DP[i][j]代表以nums1的第i位，以及nums2的第j位爲末位的重複子數組的長度，狀態轉移公式爲

    if nums1[i] == nums2[j] : dp[i][j] = dp[i-1]dp[j-1] + 1;
else:                     dp[i][j] = 0;

    所以實現如下：

    class Solution {
public:
    int findLength(vector<int>& nums1, vector<int>& nums2) {
        int n = nums1.size(), m = nums2.size();
        vector< vector<int> > dp(n, vector<int>(m,0)); 
        int res = 0;
        for(int i = 0; i < n; ++i){
            for(int j = 0; j < m; ++j){
                if(nums1[i] == nums2[j]){
                    dp[i][j] += 1;
                    if(i>0 && j >0)dp[i][j] += dp[i-1][j-1]; //注意邊界
                }
                res = max(res, dp[i][j]);
            }
        }
        return res;
    }
};

    時間複雜度： O(N * M)

    空間複雜度：O(N * M)

    在上述循環當中，我們可以發現每次循環DP[i][:]只與DP[i-1][:]有關。且DP[i][:]與DP[i-1][:]長度相同，DP[i][j]也只與DP[i-1][j-1]有關。那我們可以只用一個一位數組dp來替換原來的二維數組。

    即：

    if nums1[i] == nums2[j] : dp[j-1] + 1;
else:                     dp[j] = 0;

    且注意，由於沒有i，所以沒辦法檢查邊界。故而必須在dp數組添加一位，使得原來i==0的時候使dp[i-1][*]不會越界。在一維dp數組當中，DP[j]代表以以nums1的第i-1位，以及nums2的第j-1位爲末位的重複子數組的長度.

    實現方法如下：

    class Solution {
public:
    int findLength(vector<int>& nums1, vector<int>& nums2) {
        int n = nums1.size(), m = nums2.size();
        vector<int>  dp(m+1,0); 
        int res = 0;
        for(int i = 1; i <= n; ++i){
            for(int j = m; j > 0; --j){
                if(nums1[i-1] == nums2[j-1])
                    dp[j] = 1 + dp[j-1];
                else{
                    dp[j] = 0;
                }
                res = max(res, dp[j]);
            }
        }
        return res;
    }
};

    哈希+二分查找

    二分查找的思想如下，既如果兩個數組的最長重複子數組的長度爲k，則必然存在len < k的重複子數組，而len > k的重複子數組必然不存在，這樣就可以利用二分法來查找這個k。

    但是如何判斷nums1和nums2是否存在長度爲len的重複子數組呢。一個簡單的方法是將nums1的所有長度爲len的子數組放入到哈希表當中。然後再遍歷nums2的所有長度爲len的子數組，並此子數組查詢哈希表當中的是否存在。但是我們並不能將一個數組直接放入到哈希表當中，一種方法是將數組轉爲字符串，第二種方法是利用Rabin-Karp 算法，表達式如下，

    $hash(nums) = \displaystyle \sum^{len(nums)-1}_{i = 0}{base ^{len(nums)-(i+1)} \times nums[i]}$

    看上去蠻複雜的，其實非常簡單。就是將數組nums視爲base進制數，然後求其十進制的值。但是考慮最後取得的值會非常大，所以必須對結果進行去模。然而這樣有概率發生哈希碰撞，所以我們還是採用將數組轉化爲字符串的辦法。

    class Solution {
public:
    bool check(string & nums1, string& nums2, int len){
        unordered_set<string> s;
        for(int i = 0; i+len<= nums1.size(); ++i){
            s.insert(nums1.substr(i,len));
        }
        for(int i = 0; i+len<=nums2.size(); ++i){
            if(s.count(nums2.substr(i,len)))return true;
        }
        return false;
    }
    string toString(vector<int> & nums){
        string res;
        for(auto& n:nums)
            res += char(n);
        return res;
    }
    int findLength(vector<int>& nums1, vector<int>& nums2) {
        string n1 = toString(nums1), n2 = toString(nums2);
        int left = 0, right = min(nums1.size(),nums2.size());
        while(left < right){
            int mid = (left + right+1) / 2;
            if(check(n1,n2,mid)){
                left = mid;
            }else{
                right = mid-1;
            }
        }
        return right;
    }
};

    時間複雜度：log(min(N,M)) * (M + N)

    空間複雜度：O(N)
